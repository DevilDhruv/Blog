class LinkedList {
  constructor(value) {
    this.head = value !== undefined ? { value, next: null } : null;
    this.tail = this.head;
    this.length = this.head ? 1 : 0;
  }

  append(value) {
    const node = { value, next: null };
    if (!this.head) {
      this.head = this.tail = node;
    } else {
      this.tail.next = node;
      this.tail = node;
    }
    this.length++;
  }

  prepend(value) {
    this.head = { value, next: this.head };
    if (!this.tail) this.tail = this.head;
    this.length++;
  }

  removeHead() {
    if (!this.head) return;
    this.head = this.head.next;
    this.length--;
    if (!this.head) this.tail = null;
  }

  removeTail() {
    if (!this.head) return;
    if (this.head === this.tail) {
      this.head = this.tail = null;
      this.length = 0;
      return;
    }
    let curr = this.head;
    while (curr.next !== this.tail) {
      curr = curr.next;
    }
    curr.next = null;
    this.tail = curr;
    this.length--;
  }
}
